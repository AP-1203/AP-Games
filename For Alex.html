import pygame from pygame.locals import * import random pygame.init() screen = pygame.display.set_mode((600,500)) pygame.display.set_caption("Drawing Rectangles") colors = ["red","blue","green","yellow","orange","white"] color = "white" # set initial positions and velocity for our rectangle pos_x = 300 pos_y = 250 vel_x = 0.5 vel_y = 0.25 #our while loop, the same from other programs while True: for event in pygame.event.get(): if event.type in (QUIT, KEYDOWN): sys.exit() screen.fill("black") #move the rectangle pos_x += vel_x pos_y += vel_y #keep the rectangle on the screen if pos_x > 500 or pos_x < 0: vel_x = -vel_x color = random.choice(colors) if pos_y > 400 or pos_y < 0: vel_y = -vel_y color = random.choice(colors) #draw the rectangle pos = pos_x, pos_y, 100, 100 width = 0 pygame.draw.rect(screen, color, pos, width) pygame.display.update()
